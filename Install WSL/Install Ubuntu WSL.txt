--------Add Service Ubuntu-----------
Треба стоврити сервіс де воно саме буде запускати додаток
sudo nano /etc/systemd/system/local.spr311.com.service

sudo systemctl enable local.spr311.com.service
sudo systemctl start local.spr311.com.service
sudo systemctl status local.spr311.com.service
sudo systemctl stop local.spr311.com.service

--------Add ngixn options------------
/etc/nginx/sites-available/default

server {
server_name   local.spr311.com *.local.spr311.com;
location / {
        proxy_pass         http://localhost:4075;
        proxy_http_version 1.1;
        proxy_set_header   Upgrade $http_upgrade;
        proxy_set_header   Connection keep-alive;
        proxy_set_header   Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
    }
}

sudo systemctl restart nginx
sudo systemctl status nginx
